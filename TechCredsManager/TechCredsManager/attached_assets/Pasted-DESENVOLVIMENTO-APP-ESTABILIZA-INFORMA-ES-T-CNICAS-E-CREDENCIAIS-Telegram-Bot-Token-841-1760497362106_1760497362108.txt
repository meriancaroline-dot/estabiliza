class ValidationService {
  // Validações existentes mantidas
  
  // Novas validações:
  
  static validateName(name) {
    if (!name || !name.trim()) {
      return { valid: false, error: 'Nome é obrigatório' };
    }
    if (name.length < 2) {
      return { valid: false, error: 'Nome muito curto' };
    }
    if (name.length > 100) {
      return { valid: false, error: 'Nome muito longo (máx 100 caracteres)' };
    }
    if (!/^[a-zA-ZÀ-ÿ\s]+$/.test(name)) {
      return { valid: false, error: 'Nome contém caracteres inválidos' };
    }
    return { valid: true };
  }
  
  static validateDate(date, allowFuture = false, allowPast = true) {
    if (!date) {
      return { valid: false, error: 'Data é obrigatória' };
    }
    
    const selectedDate = new Date(date);
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    
    if (!allowFuture && selectedDate > today) {
      return { valid: false, error: 'Data não pode ser no futuro' };
    }
    
    if (!allowPast && selectedDate < today) {
      return { valid: false, error: 'Data não pode ser no passado' };
    }
    
    return { valid: true };
  }
  
  static validateTime(time) {
    if (!time) {
      return { valid: false, error: 'Horário é obrigatório' };
    }
    
    // Aceita formatos: HH:MM ou Date object
    if (time instanceof Date) {
      return { valid: true };
    }
    
    const timeRegex = /^([0-1]?[0-9]|2[0-3]):[0-5][0-9]$/;
    if (!timeRegex.test(time)) {
      return { valid: false, error: 'Horário inválido (use HH:MM)' };
    }
    
    return { valid: true };
  }
  
  static validateTextLength(text, min = 0, max = 1000) {
    const length = text ? text.trim().length : 0;
    
    if (length < min) {
      return { valid: false, error: `Mínimo ${min} caracteres` };
    }
    if (length > max) {
      return { valid: false, error: `Máximo ${max} caracteres` };
    }
    
    return { valid: true, remaining: max - length };
  }
  
  static validateUrl(url) {
    if (!url) return { valid: false, error: 'URL é obrigatória' };
    
    try {
      const urlObj = new URL(url);
      return { valid: true };
    } catch (e) {
      return { valid: false, error: 'URL inválida' };
    }
  }
  
  static sanitizeInput(input, maxLength = 1000) {
    if (typeof input !== 'string') return input;
    
    return input
      .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '')
      .replace(/<[^>]*>/g, '')
      .replace(/javascript:/gi, '')
      .replace(/on\w+\s*=/gi, '')
      .trim()
      .substring(0, maxLength);
  }
  
  static validateFile(file, maxSize = 5 * 1024 * 1024, allowedTypes = []) {
    if (!file) {
      return { valid: false, error: 'Arquivo é obrigatório' };
    }
    
    if (file.size > maxSize) {
      const maxSizeMB = maxSize / (1024 * 1024);
      return { valid: false, error: `Arquivo muito grande (máx ${maxSizeMB}MB)` };
    }
    
    if (allowedTypes.length > 0 && !allowedTypes.includes(file.type)) {
      return { valid: false, error: 'Tipo de arquivo não permitido' };
    }

class SearchService {
  static searchReminders(reminders, query, filters = {}) {
    let results = [...reminders];
    
    // Busca por texto
    if (query && query.trim()) {
      const lowerQuery = query.toLowerCase();
      results = results.filter(reminder => 
        reminder.name.toLowerCase().includes(lowerQuery) ||
        (reminder.notes && reminder.notes.toLowerCase().includes(lowerQuery))
      );
    }
    
    // Filtro por tipo
    if (filters.type && filters.type !== 'all') {
      results = results.filter(r => r.type === filters.type);
    }
    
    // Filtro por data
    if (filters.dateFrom) {
      results = results.filter(r => new Date(r.date) >= new Date(filters.dateFrom));
    }
    if (filters.dateTo) {
      results = results.filter(r => new Date(r.date) <= new Date(filters.dateTo));
    }
    
    // Filtro por status
    if (filters.status) {
      const now = new Date();
      if (filters.status === 'pending') {
        results = results.filter(r => !r.completed);
      } else if (filters.status === 'completed') {
        results = results.filter(r => r.completed);
      } else if (filters.status === 'overdue') {
        results = results.filter(r => !r.completed && new Date(r.date) < now);
      }
    }
    
    return results;
  }
  
  static sortReminders(reminders, sortBy = 'date', order = 'asc') {
    const sorted = [...reminders];
    
    sorted.sort((a, b) => {
      let comparison = 0;
      
      switch (sortBy) {
        case 'date':
          comparison = new Date(a.date) - new Date(b.date);
          break;
        case 'name':
          comparison = a.name.localeCompare(b.name);
          break;
        case 'type':
          comparison = a.type.localeCompare(b.type);
          break;
        case 'created':
          comparison = new Date(a.createdAt) - new Date(b.createdAt);
          break;
        default:
          comparison = 0;
      }
      
      return order === 'asc' ? comparison : -comparison;
    });
    
    return sorted;
  }
  
  static searchMoods(moods, filters = {}) {
    let results = [...moods];
    
    // Filtro por período do dia
    if (filters.period && filters.period !== 'all') {
      results = results.filter(m => m.period === filters.period);
    }
    
    // Filtro por humor
    if (filters.mood && filters.mood !== 'all') {
      results = results.filter(m => m.mood === filters.mood);
    }
    
    // Filtro por intervalo de datas
    if (filters.dateFrom) {
      results = results.filter(m => new Date(m.date) >= new Date(filters.dateFrom));
    }
    if (filters.dateTo) {
      results = results.filter(m => new Date(m.date) <= new Date(filters.dateTo));
    }
    
    return results;
  }
  
  static searchMessages(messages, query) {
    if (!query || !query.trim()) return messages;
    
    const lowerQuery = query.toLowerCase();
    return messages.filter(msg => 
      msg.text.toLowerCase().includes(lowerQuery)
    );
  }
  
  // Debounce para busca em tempo real
  static debounce(func, delay = 300) {
    let timeoutId;
    return (...args) => {
      clearTimeout(timeoutId);
      timeoutId = setTimeout(() => func(...args), delay);
    };
  }
}

class StatsService {
  static getMoodStats(moods, period = 'week') {
    const now = new Date();
    let startDate;
    
    switch (period) {
      case 'week':
        startDate = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
        break;
      case 'month':
        startDate = new Date(now.getTime() - 30 * 24 * 60 * 60 * 1000);
        break;
      case 'year':
        startDate = new Date(now.getTime() - 365 * 24 * 60 * 60 * 1000);
        break;
      default:
        startDate = new Date(0);
    }
    
    const filteredMoods = moods.filter(m => new Date(m.date) >= startDate);
    
    // Humor médio
    const average = filteredMoods.length > 0
      ? filteredMoods.reduce((sum, m) => sum + m.value, 0) / filteredMoods.length
      : 0;
    
    // Humor mais frequente
    const moodCounts = {};
    filteredMoods.forEach(m => {
      moodCounts[m.mood] = (moodCounts[m.mood] || 0) + 1;
    });
    const mostCommon = Object.keys(moodCounts).reduce((a, b) => 
      moodCounts[a] > moodCounts[b] ? a : b, null
    );
    
    // Sequência atual
    const streak = this.calculateMoodStreak(moods);
    
    // Dados para gráfico
    const chartData = this.prepareMoodChartData(filteredMoods, period);
    
    return {
      average: average.toFixed(1),
      mostCommon,
      streak,
      total: filteredMoods.length,
      chartData,
    };
  }
  
  static calculateMoodStreak(moods) {
    if (moods.length === 0) return 0;
    
    const sortedMoods = moods.sort((a, b) => new Date(b.date) - new Date(a.date));
    let streak = 0;
    let currentDate = new Date();
    currentDate.setHours(0, 0, 0, 0);
    
    for (const mood of sortedMoods) {
      const moodDate = new Date(mood.date);
      moodDate.setHours(0, 0, 0, 0);
      
      const diffDays = Math.floor((currentDate - moodDate) / (24 * 60 * 60 * 1000));
      
      if (diffDays === streak) {
        streak++;
      } else if (diffDays > streak) {
        break;
      }
    }
    
    return streak;
  }
  
  static prepareMoodChartData(moods, period) {
    // Preparar dados para recharts
    const data = [];
    const now = new Date();
    
    if (period === 'week') {
      for (let i = 6; i >= 0; i--) {
        const date = new Date(now.getTime() - i * 24 * 60 * 60 * 1000);
        const dayMoods = moods.filter(m => {
          const mDate = new Date(m.date);
          return mDate.toDateString() === date.toDateString();
        });
        
        const average = dayMoods.length > 0
          ? dayMoods.reduce((sum, m) => sum + m.value, 0) / dayMoods.length
          : null;
        
        data.push({
          date: date.toLocaleDateString('pt-BR', { weekday: 'short' }),
          value: average,
        });
      }
    } else if (period === 'month') {
      for (let i = 29; i >= 0; i--) {
        const date = new Date(now.getTime() - i * 24 * 60 * 60 * 1000);
        const dayMoods = moods.filter(m => {
          const mDate = new Date(m.date);
          return mDate.toDateString() === date.toDateString();
        });
        
        const average = dayMoods.length > 0
          ? dayMoods.reduce((sum, m) => sum + m.value, 0) / dayMoods.length
          : null;
        
        data.push({
          date: date.getDate().toString(),
          value: average,
        });
      }
    }
    
    return data;
  }
  
  static getHabitsStats(habits, habitCompletions) {
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    
    const todayCompletions = habitCompletions.filter(c => {
      const cDate = new Date(c.date);
      cDate.setHours(0, 0, 0, 0);
      return cDate.getTime() === today.getTime();
    });
    
    const completionRate = habits.length > 0
      ? (todayCompletions.length / habits.length) * 100
      : 0;
    
    const currentStreak = this.calculateHabitStreak(habits, habitCompletions);
    const longestStreak = this.calculateLongestStreak(habits, habitCompletions);
    const totalCompleted = habitCompletions.length;
    
    return {
      completionRate: completionRate.toFixed(0),
      currentStreak,
      longestStreak,
      totalCompleted,
    };
  }
  
  static calculateHabitStreak(habits, completions) {
    // Calcular dias consecutivos completando TODOS os hábitos
    let streak = 0;
    let currentDate = new Date();
    currentDate.setHours(0, 0, 0, 0);
    
    while (true) {
      const dayCompletions = completions.filter(c => {
        const cDate = new Date(c.date);
        cDate.setHours(0, 0, 0, 0);
        return cDate.getTime() === currentDate.getTime();
      });
      
      if (dayCompletions.length === habits.length) {
        streak++;
        currentDate.setDate(currentDate.getDate() - 1);
      } else {
        break;
      }
    }
    
    return streak;
  }
  
  static calculateLongestStreak(habits, completions) {
    // Implementação similar, mas percorrendo todo o histórico
    let maxStreak = 0;
    let currentStreak = 0;
    let currentDate = new Date();
    currentDate.setHours(0, 0, 0, 0);
    
    // Buscar data mais antiga
    if (completions.length === 0) return 0;
    
    const oldestDate = new Date(Math.min(...completions.map(c => new Date(c.date))));
    
    while (currentDate >= oldestDate) {
      const dayCompletions = completions.filter(c => {
        const cDate = new Date(c.date);
        cDate.setHours(0, 0, 0, 0);
        return cDate.getTime() === currentDate.getTime();
      });
      
      if (dayCompletions.length === habits.length) {
        currentStreak++;
        maxStreak = Math.max(maxStreak, currentStreak);
      } else {
        currentStreak = 0;
      }
      
      currentDate.setDate(currentDate.getDate() - 1);
    }
    
    return maxStreak;
  }
  
  static getRemindersStats(reminders) {
    const total = reminders.length;
    const completed = reminders.filter(r => r.completed).length;
    const pending = total - completed;
    
    const now = new Date();
    const overdue = reminders.filter(r => 
      !r.completed && new Date(r.date) < now
    ).length;
    
    const upcoming = reminders.filter(r => 
      !r.completed && new Date(r.date) >= now
    ).length;
    
    const completionRate = total > 0 ? (completed / total) * 100 : 0;
    
    return {
      total,
      completed,
      pending,
      overdue,
      upcoming,
      completionRate: completionRate.toFixed(0),
    };
  }

class SyncService {
  constructor() {
    this.queue = [];
    this.isSyncing = false;
    this.isOnline = true;
  }
  
  async initialize() {
    // Carregar fila salva
    const savedQueue = await AsyncStorage.getItem('@estabiliza:sync_queue');
    if (savedQueue) {
      this.queue = JSON.parse(savedQueue);
    }
    
    // Monitorar conectividade
    NetInfo.addEventListener(state => {
      this.isOnline = state.isConnected;
      
      if (this.isOnline && this.queue.length > 0) {
        this.processQueue();
      }
    });
  }
  
  async addToQueue(action) {
    this.queue.push({
      id: Date.now() + Math.random(),
      action,
      timestamp: new Date().toISOString(),
      retries: 0,
    });
    
    await this.saveQueue();
    
    if (this.isOnline) {
      this.processQueue();
    }
  }
  
  async processQueue() {
    if (this.isSyncing || this.queue.length === 0) return;
    
    this.isSyncing = true;
    
    const item = this.queue[0];
    
    try {
      await this.executeAction(item.action);
      
      // Remove da fila se sucesso
      this.queue.shift();
      await this.saveQueue();
      
      // Processar próximo item
      if (this.queue.length > 0) {
        setTimeout(() => {
          this.isSyncing = false;
          this.processQueue();
        }, 500);
      } else {
        this.isSyncing = false;
      }
      
    } catch (error) {
      console.error('Erro ao sincronizar:', error);
      
      item.retries++;
      
      if (item.retries >= 3) {
        // Remover após 3 tentativas
        this.queue.shift();
        Alert.alert(
          'Erro de Sincronização',
          'Não foi possível sincronizar alguns dados. Tente novamente mais tarde.'
        );
      }
      
      await this.saveQueue();
      this.isSyncing = false;
    }
  }
  
  async executeAction(action) {
    switch (action.type) {
      case 'send_message':
        return await telegramService.sendUserMessage(action.data.message, action.data.userData);
      
      case 'schedule_notification':
        return await notificationService.scheduleNotification(action.data.reminder);
      
      default:
        console.warn('Ação desconhecida:', action.type);
    }
  }
  
  async saveQueue() {
    await AsyncStorage.setItem('@estabiliza:sync_queue', JSON.stringify(this.queue));
  }
  
  getQueueSize() {
    return this.queue.length;
  }
  
  isOnlineMode() {
    return this.isOnline;
  }
}

        }
        if (data.totalEntries >= 100 && !badges.find(b => b.id === 6)?.unlocked) {
          newUnlocks.push(await this.unlockBadge(6));
        }
        break;
      
      case 'habit_completed':
        if (data.totalCompleted >= 50 && !badges.find(b => b.id === 4)?.unlocked) {
          newUnlocks.push(await this.unlockBadge(4));
        }
        break;
      
      case 'app_usage':
        const daysUsed = data.daysUsed;
        if (daysUsed >= 30 && !badges.find(b => b.id === 3)?.unlocked) {
          newUnlocks.push(await this.unlockBadge(3));
        }
        break;
      
      case 'appointment_scheduled':
        if (!badges.find(b => b.id === 5)?.unlocked) {
          newUnlocks.push(await this.unlockBadge(5));
        }
        break;
      
      case 'listening_space_used':
        if (!badges.find(b => b.id === 8)?.unlocked) {
          newUnlocks.push(await this.unlockBadge(8));
        }
        break;
      
      case 'perfect_month':
        if (data.perfectDays >= 30 && !badges.find(b => b.id === 7)?.unlocked) {
          newUnlocks.push(await this.unlockBadge(7));
        }
        break;
    }
    
    // Notificar listeners sobre novas conquistas
    newUnlocks.forEach(badge => {
      this.notifyUnlock(badge);
    });
    
    return newUnlocks;
  }
  
  async unlockBadge(badgeId) {
    const badges = await this.loadBadges();
    const badge = badges.find(b => b.id === badgeId);
    
    if (badge && !badge.unlocked) {
      badge.unlocked = true;
      badge.unlockedAt = new Date().toISOString();
      
      await this.saveBadges(badges);
      
      return badge;
    }
    
    return null;
  }
  
  async loadBadges() {
    const saved = await AsyncStorage.getItem(STORAGE_KEYS.ACHIEVEMENTS);
    if (saved) {
      return JSON.parse(saved);
    }
    
    // Retornar badges padrão (do achievements.badges array definido anteriormente)
    return achievements.badges.map(b => ({ ...b, unlocked: false }));
  }
  
  async saveBadges(badges) {
    await AsyncStorage.setItem(STORAGE_KEYS.ACHIEVEMENTS, JSON.stringify(badges));
  }
  
  onBadgeUnlock(callback) {
    this.listeners.push(callback);
  }
  
  notifyUnlock(badge) {
    this.listeners.forEach(callback => callback(badge));
  }
  
  async getProgress() {
    const badges = await this.loadBadges();
    const unlocked = badges.filter(b => b.unlocked).length;
    const total = badges.length;
    
    return {
      unlocked,
      total,
      percentage: (unlocked / total) * 100,
    };
  }
}

✅ Navegação completa (Bottom Tabs + Screens)
✅ Lembretes (CRUD completo + notificações)

✅ Hábitos (CRUD + marcação de conclusão)
✅ Chat básico (envio/recebimento via Telegram)
✅ Persistência de dados (AsyncStorage)
✅ Atualizar todos os textos conforme especificado

✅ Onboarding para novos usuários
✅ Animações básicas (fade, scale, slide)
✅ Skeleton loading screens
✅ Empty states com ilustrações
✅ Pull to refresh em listas
✅ Validação visual em tempo real
✅ Feedback com Toast ao invés de Alert
✅ Perfil do usuário com avatar
✅ Configurações básicas
